# PASS ANTI-CENTRALIZATION BILL
character_event = {
	id = laws.10000
	desc = "EVTDESCLAWS10000"
	picture = GFX_evt_speech
	is_triggered_only = yes
	
	# Liege has no regent
	option = {
		trigger = {
			NOT = {
				any_realm_character = {
					has_minor_title = title_regent
					same_liege = ROOT
				}
			}
		}
		name = "EVTOPTACONFIRM"
		liege = { character_event = { id = laws.10010 tooltip = "EVTTOOLTIPLAWS10010" } }
	}
	
	# Liege has a regent, who is in the faction and can pass the bill automatically
	option = {
		trigger = {
			any_realm_character = {
				has_minor_title = title_regent
				same_liege = ROOT
			}
		}
		name = "EVTOPTBLAWS10000"
		random_realm_character = {
			limit = {
				has_minor_title = title_regent
				same_liege = ROOT
			}
			character_event = { id = laws.10001 tooltip = "EVTTOOLTIPLAWS10001" }
		}
		clr_character_flag = flag_making_faction_ultimatum
	}
	
	# Nevermind
	option = {
		name = "EVTOPTNEVERMIND"
		clr_character_flag = flag_making_faction_ultimatum
		ai_chance = { factor = 0 }
	}
}

# A bill lowering centralization has been passed!

	id = laws.10011
	desc = "EVTDESCLAWS10011"
	picture = GFX_evt_speech
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { leads_faction = faction_anti_centralization }
			clr_character_flag = flag_making_faction_ultimatum
		}
	}
	
	option = {
		trigger = { NOT = { in_faction = faction_pro_centralization } }
		name = "EVTOPTAPPROVE"
		opinion = { who = FROM modifier = opinion_lowered_crown_authority_good }
